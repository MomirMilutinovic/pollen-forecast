>   mydata$date <- as.Date(mydata$date)

+     mutate(month = lubridate::month(date)) %>%

Adding next day

>   mydata <- mydata %>% arrange(date)
>   mydata <- mydata %>% mutate(next_day = rollapply(mydata$value, width = 2, FUN = sum, fill = NA) - mydata$value)
